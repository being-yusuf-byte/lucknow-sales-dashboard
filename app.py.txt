import streamlit as st
import pandas as pd
import plotly.express as px

# Page Setup
st.set_page_config(page_title="Lucknow Sales Dashboard", layout="wide")

# Data Load (Wahi data jo humne pehle banaya tha)
@st.cache_data
def get_data():
    # Note: Asli project mein yahan 'Lucknow_Ecommerce_Sales.xlsx' read hoga
    # Abhi ke liye hum wahi logic use kar rahe hain
    df = pd.read_excel("Lucknow_Ecommerce_Sales.xlsx")
    return df

try:
    df = get_data()

    # Sidebar Filters
    st.sidebar.header("Filter Options")
    area = st.sidebar.multiselect("Area Select Karein:", options=df["Customer_Area"].unique(), default=df["Customer_Area"].unique())
    
    df_selection = df.query("Customer_Area == @area")

    # Main Page
    st.title("??? Lucknow E-Commerce Live Dashboard")
    st.markdown("##")

    # Top Metrics
    total_sales = int(df_selection["Total_Amount"].sum())
    average_sale = round(df_selection["Total_Amount"].mean(), 2)
    
    col1, col2 = st.columns(2)
    with col1:
        st.metric("Total Revenue", f"?{total_sales:,}")
    with col2:
        st.metric("Average Order Value", f"?{average_sale}")

    st.markdown("---")

    # Charts
    left_column, right_column = st.columns(2)

    with left_column:
        fig_product = px.bar(df_selection.groupby("Product_Name").sum()[["Total_Amount"]].sort_values("Total_Amount"),
                             x="Total_Amount", y=df_selection.groupby("Product_Name").sum().sort_values("Total_Amount").index,
                             orientation="h", title="Revenue by Product")
        st.plotly_chart(fig_product, use_container_width=True)

    with right_column:
        fig_payment = px.pie(df_selection, values='Total_Amount', names='Payment_Mode', title='Payment Mode Share')
        st.plotly_chart(fig_payment, use_container_width=True)

except Exception as e:
    st.error(f"Excel file nahi mili! Pehle Excel download wala cell run karein. Error: {e}")
